## üì± Frontend Implementation

### 1. Private Key Export Component

```typescript
import { useState, useEffect } from 'react';

interface Wallet {
  id: string;
  publicKey: string;
  createdAt: string;
  isDefault: boolean;
}

interface ExportedWallet {
  id: string;
  publicKey: string;
  privateKey: string;
  createdAt: string;
}

const PrivateKeyExportComponent: React.FC = () => {
  const [wallets, setWallets] = useState<Wallet[]>([]);
  const [selectedWalletId, setSelectedWalletId] = useState<string>('');
  const [password, setPassword] = useState('');
  const [isLoading, setIsLoading] = useState(false);
  const [isExporting, setIsExporting] = useState(false);
  const [exportedWallet, setExportedWallet] = useState<ExportedWallet | null>(null);
  const [error, setError] = useState('');
  const [showPasswordModal, setShowPasswordModal] = useState(false);

  useEffect(() => {
    loadWallets();
  }, []);

  const loadWallets = async () => {
    setIsLoading(true);
    try {
      const response = await fetch('/api/user/private-keys', {
        headers: {
          'Authorization': `Bearer ${authToken}`
        }
      });

      if (response.ok) {
        const data = await response.json();
        setWallets(data.wallets);
        if (data.wallets.length > 0) {
          setSelectedWalletId(data.wallets[0].id);
        }
      } else {
        const error = await response.json();
        setError(error.message || 'Failed to load wallets');
      }
    } catch (error) {
      setError('Network error loading wallets');
    } finally {
      setIsLoading(false);
    }
  };

  const handleExport = async () => {
    if (!selectedWalletId || !password) {
      setError('Please select a wallet and enter your password');
      return;
    }

    setIsExporting(true);
    setError('');

    try {
      const response = await fetch('/api/user/private-keys/export', {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          'Authorization': `Bearer ${authToken}`
        },
        body: JSON.stringify({
          walletId: selectedWalletId,
          password: password
        })
      });

      if (response.ok) {
        const data = await response.json();
        setExportedWallet(data.wallet);
        setPassword('');
        setShowPasswordModal(false);
      } else {
        const error = await response.json();
        setError(error.message || 'Failed to export private key');
      }
    } catch (error) {
      setError('Network error exporting private key');
    } finally {
      setIsExporting(false);
    }
  };

  const copyToClipboard = async (text: string) => {
    try {
      await navigator.clipboard.writeText(text);
      // Show success message
    } catch (error) {
      console.error('Failed to copy to clipboard:', error);
    }
  };

  const downloadPrivateKey = () => {
    if (!exportedWallet) return;

    const content = `Solana Wallet Private Key Export
Generated: ${new Date().toISOString()}

Wallet Information:
- Public Key: ${exportedWallet.publicKey}
- Private Key: ${exportedWallet.privateKey}
- Created: ${exportedWallet.createdAt}

‚ö†Ô∏è SECURITY WARNING:
- Keep this private key secure and never share it
- Anyone with this private key can access your wallet
- Store it in a safe place offline
- Consider using a hardware wallet for better security

This file was generated by Yappysol on ${new Date().toLocaleString()}`;

    const blob = new Blob([content], { type: 'text/plain' });
    const url = URL.createObjectURL(blob);
    const a = document.createElement('a');
    a.href = url;
    a.download = `solana-wallet-${exportedWallet.publicKey.slice(0, 8)}.txt`;
    document.body.appendChild(a);
    a.click();
    document.body.removeChild(a);
    URL.revokeObjectURL(url);
  };

  const resetExport = () => {
    setExportedWallet(null);
    setPassword('');
    setError('');
  };

  if (isLoading) {
    return (
      <div className="private-key-export">
        <div className="loading">
          <div className="spinner"></div>
          <p>Loading wallets...</p>
        </div>
      </div>
    );
  }

  if (exportedWallet) {
    return (
      <div className="private-key-export">
        <div className="export-success">
          <h3>üîê Private Key Exported Successfully</h3>
          
          <div className="wallet-info">
            <div className="info-item">
              <label>Public Key:</label>
              <div className="key-display">
                <code>{exportedWallet.publicKey}</code>
                <button 
                  onClick={() => copyToClipboard(exportedWallet.publicKey)}
                  className="copy-button"
                >
                  üìã Copy
                </button>
              </div>
            </div>

            <div className="info-item">
              <label>Private Key:</label>
              <div className="key-display">
                <code className="private-key">{exportedWallet.privateKey}</code>
                <button 
                  onClick={() => copyToClipboard(exportedWallet.privateKey)}
                  className="copy-button"
                >
                  üìã Copy
                </button>
              </div>
            </div>
          </div>

          <div className="security-warning">
            <h4>‚ö†Ô∏è Security Warning</h4>
            <ul>
              <li>Keep this private key secure and never share it</li>
              <li>Anyone with this private key can access your wallet</li>
              <li>Store it in a safe place offline</li>
              <li>Consider using a hardware wallet for better security</li>
            </ul>
          </div>

          <div className="export-actions">
            <button onClick={downloadPrivateKey} className="download-button">
              üì• Download as File
            </button>
            <button onClick={resetExport} className="secondary-button">
              Export Another Wallet
            </button>
          </div>
        </div>
      </div>
    );
  }

  return (
    <div className="private-key-export">
      <h3>Export Private Keys</h3>
      <p>Export your wallet's private key for backup or use in other wallets.</p>

      {error && (
        <div className="error-message">
          ‚ùå {error}
        </div>
      )}

      <div className="wallet-selection">
        <label>Select Wallet:</label>
        <select 
          value={selectedWalletId} 
          onChange={(e) => setSelectedWalletId(e.target.value)}
          disabled={wallets.length === 0}
        >
          {wallets.map(wallet => (
            <option key={wallet.id} value={wallet.id}>
              {wallet.publicKey} {wallet.isDefault ? '(Default)' : ''}
            </option>
          ))}
        </select>
      </div>

      {wallets.length === 0 ? (
        <div className="no-wallets">
          <p>No wallets found. Please create a wallet first.</p>
        </div>
      ) : (
        <div className="export-section">
          <button 
            onClick={() => setShowPasswordModal(true)}
            className="export-button"
          >
            üîê Export Private Key
          </button>
        </div>
      )}

      {/* Password Modal */}
      {showPasswordModal && (
        <div className="password-modal-overlay">
          <div className="password-modal">
            <h4>Enter Your Password</h4>
            <p>Please enter your password to export the private key:</p>
            
            <input
              type="password"
              value={password}
              onChange={(e) => setPassword(e.target.value)}
              placeholder="Enter your password"
              className="password-input"
            />

            <div className="modal-actions">
              <button 
                onClick={() => {
                  setShowPasswordModal(false);
                  setPassword('');
                  setError('');
                }}
                className="cancel-button"
              >
                Cancel
              </button>
              <button 
                onClick={handleExport}
                disabled={!password || isExporting}
                className="confirm-button"
              >
                {isExporting ? 'Exporting...' : 'Export'}
              </button>
            </div>
          </div>
        </div>
      )}
    </div>
  );
};

export default PrivateKeyExportComponent;
```

## üîê Security Features

### 1. Password Protection
- ‚úÖ **Password required** to export private keys
- ‚úÖ **Server-side verification** of password
- ‚úÖ **No private keys** returned without authentication

### 2. User Authorization
- ‚úÖ **Wallet ownership verification** - users can only export their own wallets
- ‚úÖ **JWT authentication** required for all endpoints
- ‚úÖ **Access control** prevents unauthorized access

### 3. Secure Display
- ‚úÖ **Private key highlighting** for visibility
- ‚úÖ **Copy to clipboard** functionality
- ‚úÖ **Download as file** option
- ‚úÖ **Security warnings** prominently displayed

## üìã API Endpoints Summary

### GET /api/user/private-keys
**Purpose**: List user's wallets (without private keys)
**Response**:
```json
{
  "success": true,
  "wallets": [
    {
      "id": "wallet123",
      "publicKey": "ABC123...",
      "createdAt": "2024-01-01T00:00:00Z",
      "isDefault": true
    }
  ]
}
```

### POST /api/user/private-keys/export
**Purpose**: Export private key (requires password)
**Request**:
```json
{
  "walletId": "wallet123",
  "password": "userpassword"
}
```
**Response**:
```json
{
  "success": true,
  "wallet": {
    "id": "wallet123",
    "publicKey": "ABC123...",
    "privateKey": "XYZ789...",
    "createdAt": "2024-01-01T00:00:00Z"
  },
  "message": "Private key exported successfully. Keep it secure!"
}
```
